## 6. Smoke 测试脚本（伪代码）

以下为 Python 脚本框架，用于快速验证 Flet UI 进程启动是否正常：

```python
#!/usr/bin/env python3
"""Smoke test: 启动 Flet UI 并验证进程是否正常运行。"""
import subprocess
import sys

def smoke_test_flet_ui(timeout_sec: int = 10) -> bool:
    """启动 pyclaw ui（或 flet run），检查进程在 timeout 内是否保持运行。"""
    # 方式 1: pyclaw ui
    cmd = [sys.executable, "-m", "pyclaw", "ui"]
    # 方式 2: flet run
    # cmd = ["flet", "run", "src/pyclaw/ui/app.py"]

    proc = subprocess.Popen(
        cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        cwd="/path/to/openclaw-py",  # 替换为实际项目路径
    )

    try:
        # 等待 timeout 秒，若进程退出则返回 False
        ret = proc.wait(timeout=timeout_sec)
        return False  # 提前退出表示异常
    except subprocess.TimeoutExpired:
        proc.terminate()
        try:
            proc.wait(timeout=5)
        except subprocess.TimeoutExpired:
            proc.kill()
        return True  # 超时未退出表示进程正常运行

if __name__ == "__main__":
    ok = smoke_test_flet_ui(timeout_sec=8)
    sys.exit(0 if ok else 1)
```
